import requests import time def obtenir_prix_btc(): # URL de l'API publique de Binance url = "https://api.binance.com/api/v3/ticker/price?symbol=BTCUSDT" try: response = requests.get(url) data = response.json() symbole = data['symbol'] prix = data['price'] print(f"Actif : {symbole} | Prix actuel : {prix} USD") except Exception as e: print(f"Erreur lors de la récupération : {e}") # Boucle pour tester la compétence "Temps Réel" (10 itérations) print("--- Lancement du test de flux de marché ---") for i in range(10): obtenir_prix_btc() time.sleep(2) # Attend 2 secondes entre chaque mise à jour
